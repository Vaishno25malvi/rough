import javax.swing.*;
import javax.swing.tree.DefaultMutableTreeNode;

public class HtmlTreeExample {
    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            JTree tree = new JTree(createNodes());
            tree.setCellRenderer(new HtmlTreeCellRenderer());

            JFrame frame = new JFrame("HTML JTree Example");
            frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            frame.add(new JScrollPane(tree));
            frame.setSize(400, 300);
            frame.setVisible(true);
        });
    }

    private static DefaultMutableTreeNode createNodes() {
        DefaultMutableTreeNode root = new DefaultMutableTreeNode("Root");

        // Example HTML content with ANSI codes parsed
        String htmlContent1 = "This is <b>bold</b> text";
        String htmlContent2 = "This is <i>italic</i> text";

        root.add(new DefaultMutableTreeNode(htmlContent1));
        root.add(new DefaultMutableTreeNode(htmlContent2));

        return root;
    }
}

import javax.swing.*;
import javax.swing.tree.DefaultTreeCellRenderer;
import java.awt.*;

public class HtmlTreeCellRenderer extends DefaultTreeCellRenderer {
    @Override
    public Component getTreeCellRendererComponent(JTree tree, Object value,
                                                  boolean selected, boolean expanded,
                                                  boolean leaf, int row, boolean hasFocus) {
        Component c = super.getTreeCellRendererComponent(tree, value, selected, expanded, leaf, row, hasFocus);

        if (c instanceof JLabel) {
            JLabel label = (JLabel) c;
            // Assume value.toString() returns the HTML content with parsed ANSI codes
            label.setText("<html>" + value.toString() + "</html>");
        }

        return c;
    }
}
